\section{FrBu II.2 7}
\questionBox{
Let $[a,b]$ and $[c,d]$ ( $a < b$ and $c < d$ ) be compact intervals in $\mathbb{R}$. \\
\textit{Show:} there is an affine map
\begin{alignat*}{1}
    \varphi: [a,b] & \rightarrow [c,d], \\
    t & \mapsto \alpha t + \beta,
\end{alignat*}
with $\varphi(a)=c$ and $\varphi(b)=d$.
}

\answerBox{}
