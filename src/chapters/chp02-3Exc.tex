\section{Exercises}


\subsection{Question 1 a).}
\noindent
Find all elements $A$ of $\Gamma$ which commute with $S$.

\subsection*{Answer}
\noindent
We answer this by proving the following proposition.

\begin{proposition}
    $1=1$
\end{proposition}

\begin{proof}
    Assume that:

    \begin{align*}
        1 &= 1
    \end{align*}

\end{proof}
